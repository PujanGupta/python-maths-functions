```
from pythonarithmeticfunctions import *

triangleArea(base,height)# -> returns area of triangle with base and height
triangleArea(2,3)#         -> 3

checkIfPrime(integer)# -> returns boolean whether passed integer is prime or not
checkIfPrime(2)#       -> returns True
checkIfPrime(3)#       -> returns False

primeCompositeInRange(rangeLower,rangeUpper)#both params inclusive
primeCompositeInRange(rangeLower,rangeUpper)#returns dict {'primes':[primes],'composites':[composites]}

reversme(string)# -> returns reversed string

checkPalindrome(palindrome)# -> returns boolean is passed string is palindrome
checkPalindrome("meowwoem")# -> returns True
checkPalindrom("meowoomeow")# -> returns False

generatePalindromes(count)# -> generates count number of palindromes
generatePalindromes(110)# -> generates 110 palindromes

palindromicPyramidGenerator(rows)# -> generates a palindromic pyramid with rows rows
palindromicPyramidGenerator(13)# -> generates a palindromic pyramid with 13 rows

getAllPalindromesinRange(start, end)# -> returns all palindromes in range(start, end+1)

findLCM(x, y)# -> returns the lowest common multiple of x and y
findLCM(8,16)# -> returns 16, the lowest common multiple of 8 and 16

checkIfPalindrome(s)# -> returns boolean if s is a palindrome

findHCF(x, y)# -> returns the highest common factor of x and y
findHCF(6,12)# -> returns the highest common factor of 6 and 12, 6

findFactorsOf(n)# -> returns all the factors of n
```
